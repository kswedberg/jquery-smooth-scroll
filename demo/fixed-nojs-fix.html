$(document).ready(function() {
  $('button').one('click', function() {

    // Start stupid hack
    var $stupid = $('<div></div>')
      .height(1)
      .hide()
      .appendTo('body');

    $('ul.mainnav').on('click', 'a', function() {
      $stupid.show();
      setTimeout(function() {
        $stupid.hide();
      }, 10);
    });
    // end stupid hack

  });
});
